RELATIVITY
	SUBTRACTION
		v0-v1
		
		RESULTING VECTOR:
			v0's position "according" (relative to) v1
			
			PROPERTIES:
				DIRECTION: start from v1, face v0
				MAGNITUDE: distance between the two
	
	TO OBJECT POSITION (both vectors must represent world coordinates)
		O2RelativeO1 = O2Pos-O1Pos
		
	TO WORLD POSITION
		WorldRelativeO1 = worldPos - O1Pos
		O2Pos = O2RelativeO1 - WorldRelativeO1
		
		IN OTHER WORDS
			O2Pos = O2RelativeO1 - (worldPos - O1Pos)
		WHICH MEANS
			O2Pos == O2Pos - O1Pos - (worldPos - O1Pos)
		WHICH IS TRUE BECAUSE
			O2Pos == O2Pos - O1Pos - worldPos + O1Pos
			= O2Pos - worldPos - O1Pos + O1Pos
			= O2Pos - 0
			= O2Pos
